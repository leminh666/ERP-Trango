# =============================================================================
# Frontend Environment Configuration - ERP Trango v3.0
# =============================================================================
#
# INSTRUCTIONS:
# 1. Copy this file to: apps/web/.env.local
# 2. Fill in your values  
# 3. Restart dev server
#
# =============================================================================
# SCENARIO 1: DESKTOP DEVELOPMENT (localhost)
# =============================================================================
# Backend chạy trên localhost:4000
# Frontend chạy trên localhost:3000
# Dùng proxy mode để tránh CORS
NEXT_PUBLIC_API_URL=http://localhost:4000
NEXT_PUBLIC_USE_PROXY=true
# =============================================================================
# SCENARIO 2: LAN / MOBILE ACCESS
# =============================================================================
# 1. Backend phải bind tới LAN IP: `npm run start:dev` (không dùng localhost)
# 2. Frontend cũng chạy trên LAN IP
# 3. Mobile truy cập: http://<LAN_IP>:3000
#
# Cấu hình với LAN IP:
# NEXT_PUBLIC_API_URL=http://192.168.1.12:4000
# NEXT_PUBLIC_USE_PROXY=true
# =============================================================================
# SCENARIO 3: DOCKER
# =============================================================================
# Backend trong container, frontend ở host
# NEXT_PUBLIC_API_URL=http://localhost:4000
# NEXT_PUBLIC_USE_PROXY=true
# =============================================================================
# Optional Settings
# =============================================================================
NEXT_PUBLIC_API_TIMEOUT=15000
# NEXT_PUBLIC_ASSET_URL=http://localhost:4000
NEXT_PUBLIC_ENABLE_LOGGING=true
